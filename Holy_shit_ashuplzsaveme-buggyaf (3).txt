; ========= Falklands ABM (Terrain + Land Troops) =========
; Requires: NetLogo GIS extension, ASCII grid DEM at 1km cellsize
; DEM: Falklands_DEM_1km.asc (EPSG:32721, cellsize 1000m -> 1 patch = 1 km)

extensions [ gis ]

globals [
  night?
  dem
  dem-path
  cellsize-m          ; meters per patch (from ASCII header)
  landing-patch       ; staging area for British
  stanley-patch       ; objective / urban area
]

breed [ british-troops british-troop ]
breed [ argentine-troops argentine-troop ]

turtles-own [
  morale
  fatigue
  ammo
  training
  cohesion
  supply-level
  defensive-bonus
  attack-bonus
  isolated?
  entrenched?
  unit-type
  objective           ; patch (or nobody)
]

patches-own [
  elevation
  slope
  terrain             ; "water" "moorland" "bog" "ridge" "urban"
  ridge?
  stanley?
  landing?
]

;; ================== NUMERIC SAFETY HELPERS ==================

to-report is-valid? [x]
  ; true only for a normal number (not NaN, not crazy huge)
  report (is-number? x) and (x = x) and (abs x < 100000)
end

to-report safe-elev-of [p fallback]
  if p = nobody [ report fallback ]
  let v [elevation] of p
  if not is-valid? v [ report fallback ]
  report v
end

to-report clamp [x lo hi]
  report max list lo (min list hi x)
end

; -------------------- TUNABLE CONSTANTS --------------------
to set-default-parameters
  set dem-path (word user-home "/Desktop/falklands_gis/processed/Falklands_DEM_1km.asc")
end

to-report user-home
  report (word "/Users/" user-name)
end

to-report user-name
  report "elliottbarthel"
end

; -------------------- SETUP --------------------
to setup
  clear-all
  set landing-patch nobody
  set stanley-patch nobody
  set-default-parameters

  ; 1) Load DEM and resize world to match raster
  set dem gis:load-dataset dem-path
  let w gis:width-of dem
  let h gis:height-of dem

  resize-world 0 (w - 1) 0 (h - 1)
  set-patch-size 10

  ; 2) Apply raster to patches
  ask patches [
    set elevation 0
    set slope 0
    set terrain "water"
    set ridge? false
    set stanley? false
    set landing? false
  ]
  gis:apply-raster dem elevation

  ; 3) Determine cellsize in meters (assumes 1km ASC; keep explicit)
  set cellsize-m 1000

  ; 4) Classify terrain from elevation + slope
  classify-water-land
  compute-slope
  classify-terrain

  ; 5) Define key locations
  define-key-locations

  ; 6) Create troops
  spawn-troops

  set night? false
  reset-ticks
end

; -------------------- TERRAIN --------------------
to classify-water-land
  ask patches [
    ifelse elevation <= 0
    [ set terrain "water" ]
    [ set terrain "moorland" ]
  ]
end

to compute-slope
  ; Preconditions (observer level)
  if (not is-number? cellsize-m) or cellsize-m <= 0 [
    user-message (word "compute-slope: invalid cellsize-m = " cellsize-m " (must be > 0)")
    stop
  ]

  ask patches with [ terrain != "water" ] [

    set slope 0

    ; Elevation must be valid
    if not is-valid? elevation [ stop ]

    let elev-self elevation

    ; Get neighbor elevations safely (NOTE THE PARENTHESES)
    let elev-east  safe-elev-of (patch-at  1  0) elev-self
    let elev-west  safe-elev-of (patch-at -1  0) elev-self
    let elev-north safe-elev-of (patch-at  0  1) elev-self
    let elev-south safe-elev-of (patch-at  0 -1) elev-self

    ; Central differences (rise/run)
    let slope-x (elev-east - elev-west) / (2 * cellsize-m)
    let slope-y (elev-north - elev-south) / (2 * cellsize-m)

    if not is-valid? slope-x [ set slope 0 stop ]
    if not is-valid? slope-y [ set slope 0 stop ]

    let ax abs slope-x
    let ay abs slope-y
    let max-comp max list ax ay

    if (not is-valid? max-comp) or max-comp = 0 [
      set slope 0
      stop
    ]

    let nx slope-x / max-comp
    let ny slope-y / max-comp

    if (is-valid? nx) and (is-valid? ny) [
      set slope 100 * max-comp * sqrt (nx * nx + ny * ny)
    ]
    if not ((is-valid? nx) and (is-valid? ny)) [
      set slope 0
    ]
  ]
end

to classify-terrain
  ask patches [
    set ridge? false
    set stanley? false
    set landing? false

    if terrain = "water" [ stop ]

    if slope < 2.0 [
      set terrain "moorland"
    ]
    if slope >= 2.0 and slope < 6.0 [
      set terrain "bog"
    ]
    if slope >= 6.0 [
      set terrain "ridge"
      set ridge? true
    ]
  ]
  recolor
end

to recolor
  ask patches [
    if terrain = "water"    [ set pcolor blue ]
    if terrain = "moorland" [ set pcolor green ]
    if terrain = "bog"      [ set pcolor brown ]
    if terrain = "ridge"    [ set pcolor gray ]
    if terrain = "urban"    [ set pcolor white ]
  ]
end

to define-key-locations
  let east-band patches with [terrain != "water" and pxcor > (max-pxcor - 30)]
  let west-band patches with [terrain != "water" and pxcor < (min-pxcor + 60)]

  if any? east-band [
    set stanley-patch min-one-of east-band [ (slope * 3) + distancexy max-pxcor (0.5 * (min-pycor + max-pycor)) ]
    ask stanley-patch [
      set stanley? true
      set terrain "urban"
    ]
  ]

  if any? west-band [
    set landing-patch min-one-of west-band [ (slope * 3) + distancexy min-pxcor (0.65 * max-pycor) ]
    ask landing-patch [
      set landing? true
      set terrain "moorland"
    ]
  ]

  recolor
end

; -------------------- TROOPS --------------------
to spawn-troops
  ask turtles [ die ]

  let land patches with [ terrain != "water" ]
  if not any? land [
    user-message "No land patches found (check DEM classification)."
    stop
  ]

create-british-troops 120 [
  set color blue
  set morale 90
  set fatigue 10
  set ammo 100
  set training 85
  set cohesion 85
  set supply-level 90
  set attack-bonus 15
  set defensive-bonus 5
  set isolated? false
  set entrenched? false
  set unit-type "regular"
  set objective stanley-patch

  if landing-patch != nobody [
    let nearby-patches ([patches in-radius 8] of landing-patch)
    let valid-landing nearby-patches with [ terrain != "water" ]

    ifelse any? valid-landing [
      move-to one-of valid-landing
    ] [
      move-to one-of land
    ]
  ]
  if landing-patch = nobody [
    move-to one-of land
  ]
]

  ; Argentines: spawn around Stanley / ridges if available
  create-argentine-troops 120 [
    set color red
    set morale 60
    set fatigue 20
    set ammo 80
    set training 45
    set cohesion 50
    set supply-level 60
    set defensive-bonus 20
    set attack-bonus 5
    set isolated? false
    set entrenched? true
    set unit-type "regular"
    set objective nobody

    ifelse stanley-patch != nobody [
      let defensive-area ([patches in-radius 25] of stanley-patch)
      set defensive-area defensive-area with [ terrain != "water" ]

      if any? defensive-area [
        let ridge-candidates defensive-area with [ terrain = "ridge" ]
        ifelse any? ridge-candidates
        [ move-to one-of ridge-candidates ]
        [ move-to one-of defensive-area ]
      ]
      if not any? defensive-area [
        move-to one-of land
      ]
    ]
    [
      move-to one-of land
    ]
  ]
end

; -------------------- TIME / VISIBILITY --------------------
to update-time
  ifelse (ticks mod 10) < 5
  [ set night? true ]
  [ set night? false ]
end

; -------------------- MOVEMENT --------------------
to british-move
  ask british-troops [
    if morale <= 20 [ stop ]
    if fatigue >= 90 [ stop ]

    let speed patches-per-tick-british
    if night? [ set speed speed * 0.85 ]
    if supply-level < 40 [ set speed speed * 0.70 ]
    set speed speed * terrain-speed-mult terrain

    ifelse objective != nobody [ face objective ]
    [ rt (random 25 - random 25) ]

    move-land-safe speed
    set fatigue fatigue + (1 + (1 - terrain-speed-mult terrain) * 3)
  ]
end

to argentine-move
  ask argentine-troops [
    if entrenched? [ stop ]
    if morale <= 25 [ stop ]
    if fatigue >= 90 [ stop ]

    let speed patches-per-tick-argentine
    if night? [ set speed speed * 0.90 ]
    if supply-level < 40 [ set speed speed * 0.70 ]
    set speed speed * terrain-speed-mult terrain

    rt (random 30 - random 30)
    move-land-safe speed
    set fatigue fatigue + 2
  ]
end

to-report patches-per-tick-british
  report 1
end

to-report patches-per-tick-argentine
  report 0.7
end

to-report terrain-speed-mult [t]
  if t = "moorland" [ report 1.0 ]
  if t = "bog"      [ report 0.65 ]
  if t = "ridge"    [ report 0.75 ]
  if t = "urban"    [ report 0.85 ]
  report 0.0
end

to move-land-safe [dist]
  let steps ceiling (dist * 4)
  if steps < 1 [ set steps 1 ]
  let step (dist / steps)

  repeat steps [
    let next patch-ahead step
    if next = nobody [ stop ]
    if [terrain] of next = "water" [
      rt 30
      set next patch-ahead step
      if next = nobody [ stop ]
      if [terrain] of next = "water" [ lt 60 ]
    ]
    set next patch-ahead step
    if next != nobody and [terrain] of next != "water" [
      fd step
    ]
  ]
end

; -------------------- COMBAT --------------------
to british-attack
  ask british-troops [
    if ammo <= 0 [ stop ]
    let target one-of argentine-troops in-radius 2
    if target = nobody [ stop ]

    let combat-power (training + morale + attack-bonus - fatigue)
    let defense-power ([training] of target + [defensive-bonus] of target + (ifelse-value [entrenched?] of target [10] [0]))

    let p clamp ((combat-power - defense-power) + 50) 5 95
    if random-float 100 < p [ ask target [ die ] ]

    set ammo ammo - 4
    set fatigue fatigue + 4
  ]
end

to argentine-attack
  ask argentine-troops [
    if ammo <= 0 [ stop ]
    if morale < 30 [ stop ]
    let target one-of british-troops in-radius 2
    if target = nobody [ stop ]

    let combat-power (training + morale + attack-bonus - fatigue)
    let defense-power ([training] of target + [defensive-bonus] of target)

    let p clamp ((combat-power - defense-power) + 45) 5 90
    if random-float 100 < p [ ask target [ die ] ]

    set ammo ammo - 4
    set fatigue fatigue + 5
  ]
end

; -------------------- SUPPLY / MORALE --------------------
to update-supply
  ask british-troops [ set supply-level supply-level - 0.4 ]
  ask argentine-troops [ set supply-level supply-level - 0.7 ]

  ask turtles [
    if supply-level < 0 [ set supply-level 0 ]
    if supply-level = 0 [
      set ammo ammo - 1
      set morale morale - 2
    ]
    if ammo < 0 [ set ammo 0 ]
  ]
end

to update-morale
  ask turtles [
    if isolated? [ set morale morale - 3 ]
    if supply-level < 35 [ set morale morale - 1.5 ]
    if fatigue > 75 [ set morale morale - 2 ]
    set morale clamp morale 0 100
    if morale < 20 [ set color gray ]
  ]
end

; -------------------- END CONDITIONS --------------------
to check-collapse
  if count argentine-troops with [morale > 20] < 20 [
    ask argentine-troops [
      set morale 0
      set color white
    ]
    stop
  ]
end

; -------------------- MAIN LOOP --------------------
to go
  update-time
  british-move
  argentine-move
  british-attack
  argentine-attack
  update-supply
  update-morale
  check-collapse
  tick
end
